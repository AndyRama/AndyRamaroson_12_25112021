import propTypes from 'prop-types'
import './User.scss'
import avatar from './../../../user.svg'

/**

 * @param {string} Data.firstName - FirstName user

function User({ id, name }) {
  return (

}

User.propTypes = {

  name: propTypes.string.isRequired,
}

export default User
