/**
 *
 * @param {string} urlId  => (/user/id)
 * @param {string} props.id
 * @param {object} data
 * @returns {object} data > send to formated
 */

export async function getUser(id) {
  try {
    const {
      data: { data },
    } = await axios.get(`http://localhost:3000/user/${id}`)
    // console.log(data);
    return data
  } catch (error) {
    console.error(error)
  }
}
